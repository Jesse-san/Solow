class Solow_class: # country is an ISO code and should be a string
                   # Year is an integer value from 1954 - 2019

    def __init__ (self, country, year):
        self.country = country
        self.year = year
        self.df_country = None # Main Dataframe with ALL Country Data
        self.dict = {} # dictionary with calculations and subsetted data
        # Workable Variables
        # These variables are used for comparing data
        self.rho = 0.0
        self.k = None # used to pass through capital value for given year 
        self.yhat = None # used to predict data
        self.a = None
        self.tfp = None
    
    def get_data(self):
        '''
        Retrieves data from pwt100.xlsx and creates workable dataframes
        '''
        # retrieving data and making it more manageable
        xls = pd.read_excel('pwt100.xlsx', 'Data')
        self.df_country = xls[xls["countrycode"] == self.country]
        self.df_country.index = self.df_country['year'] # makes indexing much easier!
        self.df_country = self.df_country.dropna(subset = 'rtfpna')
        
    def preliminary(self):
        self.dict['y'] = self.df_country['rgdpna'] / self.df_country['emp'] # gdp per cap
        self.dict['k'] = self.df_country['rnna'] / self.df_country['emp']
        self.dict['tfp'] = self.df_country['rtfpna']    
        self.k = self.dict['k'].loc[self.year] # creating k value  
        
    def tfp_regression(self):
        lhv = np.log(self.df_country['rtfpna'])
        rhv = np.arange(len(lhv))
        rhv = sm.add_constant(rhv)
        
        lr = sm.OLS(lhv, rhv).fit()
        self.dict['resids'] = lhv - lr.predict(rhv)
        #self.dict['n'] = np.exp(lr.params[0])
        #self.dict['g'] = np.exp(lr.params[1])
        print(lr.summary())
        
    def compute_parameters(self):
        self.dict['delta'] = self.df_country['delta'].mean()
        self.dict['s'] =  self.df_country['csh_i'].mean()
        self.dict['a_k'] = 1 - self.df_country['labsh'].mean()
        self.dict['n'] = (np.log(self.df_country.loc[2019, 'emp']) - np.log(self.df_country.loc[1954, 'emp'])) / len(self.df_country)
        self.dict['g'] = (np.log(self.df_country.loc[2019, 'rtfpna']) - np.log(self.df_country.loc[1954, 'rtfpna'])) / len(self.df_country)
        self.dict['yhat'] = self.intensive(self.k, A = 1)
        self.dict['A'] = self.dict['y'].loc[2017] / self.dict['yhat']
        return pd.DataFrame(self.dict)
        
    def intensive(self, k, A):
        if self.rho == 0:
            f = lambda k: A * k ** self.dict['a_k']
        else:
            f = lambda k: A * ((self.dict['a_k'] * k ** self.rho) + (1 - self.dict['a_k'])) ** (1/self.rho)
        return f(k)
    
    def compare_data(self): # compares actual data with estimates
        self.tfp = self.df_country['rtfpna'].dropna()
        self.a = 1-self.df_country['labsh']
        self.yhat = (self.tfp * self.dict['k'] ** self.a).dropna()
        self.yhat = self.dict['A']*self.yhat
        plt.title('Per capita GDP and Model prediction')
        plt.ylabel('logarithm of 2017 US $ per worker')
        np.log(self.dict['y']).plot(color = 'steelblue')
        np.log(self.yhat).plot(color='red')
        plt.legend(['Data','Prediction'])
        plt.show()
        
    def solow(self, k):
                sfk = self.dict['n'] + self.dict['g'] + (self.dict['n'] * self.dict['g']) + self.dict['delta']
                thinning = (k * sfk)
                thickening = self.dict['s'] * self.intensive(self.k , self.dict['A'])
                return thickening - thinning

    def my_bisection(self, a, b, tol): 
        if np.sign(self.solow(a)) == np.sign(self.solow(b)):
               raise Exception("The scalars a and b do not bound a root")
               
        m = (a + b) / 2       
        
        if np.abs(self.solow(m)) < tol:
            return m
        elif np.sign(self.solow(a)) == np.sign(self.solow(m)):
            return self.my_bisection(a = m, b = b, tol = tol)
        elif np.sign(self.solow(b)) == np.sign(self.solow(m)):
            return self.my_bisection(a = a, b = m, tol = tol)    
        
    def steady_state(self):
        kbar = self.my_bisection(a = 0.1, b = 1000000, tol = 1e-6)
        print('The steady-state capital stock per efficiency unit of worker is:\n',np.around(kbar, 0))
        ybar = self.intensive(kbar, self.dict['A'])
        print('Steady-state GDP per efficiency unit of worker is:\n',np.around(ybar, 0))
        
        x = symbols('x')
        rbar = diff(self.intensive(x, self.dict['A'])).subs(x, kbar)  #This is tricky syntax
        print("The steady-state rentals rate is:\n", np.around(float(rbar),3))
        print("In steady-state, capital's share is:\n", np.around(float(rbar)*kbar/ybar,2))
        
        # half life
        b = ( 1 - self.dict['delta'] + self.dict['s'] * float(rbar)) / (1 + self.dict['n'] + self.dict['g'] + self.dict['n'] * self.dict['g'])
        print("b is", np.around(b,3))
        half_life = -np.log(2)/np.log(b)
        print("The half-life of the system is:\n",np.around(half_life,1))
        
    def compare_usa(self): # compares USA data with a given country
        var_names = ['GDPC1']
        start = datetime.datetime (1954, 1, 1)
        end = datetime.datetime (2019, 10, 1)
        df_GDP = pdr.DataReader(var_names, 'fred', start,end)
        Y_cycle, Y_trend = sm.tsa.filters.hpfilter(np.log(df_GDP['GDPC1']), 1600)
        df_GDP["Y_Cycle"] = Y_cycle
        x = df_GDP.groupby(df_GDP.index.year)['Y_Cycle'].mean()
        plt.title('Solow residuals')
        plt.ylabel('Percentage deviations')
        x.plot(color = "red")
        resids = self.dict['resids']
        resids.plot(color = 'steelblue')
        plt.legend(['USA',self.country])
        plt.show()
        r = np.corrcoef(x, resids)[0,1]
        print('The correlation coefficient is:\n',r)
        n = len(x)
        tstat = r*np.sqrt(n-2)/np.sqrt(1-r**2)
        tstat 
        p_value = scipy.stats.t.sf(abs(tstat), df = n-2)
        print('Its p-value is:\n',p_value)

Comment:¶

The idea with this analysis is to see the relationship, if any, between TFP and growth. The correlation between the US solow residuals and US growth is statistically significant (p < 0.01), which suggests that own-GDP and TFP are linked.

However, we see some interesting results comparing the solow residuals of other countries to US growth. The other countries we chose for comparison, Germany, Japan, and Mexico, show a wider range of deviation in TFP than does the US. These countries see a spike in real TFP during the post-war decades of the 50s and 60s, staying above US GDP growth but typically starting to taper off in the 90s. Interestingly, we observe that Mexico gets more closely linked to the movement of US GDP starting in the mid 90s, which could be attributable to a particular trade agreement that went into effect in 1994.
